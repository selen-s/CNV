def implement(cls, u, gb): 
        """
        gb is GRAVITY MEASUREMENT in the BODY FRAME
        u is PREDICTED DYNAMICS
        """
        # compute the origin acceleration/gravity
        u0 = cls.psi(np.linalg.inv(cls.X0), gb)
        R =  cls.X[0:3, 0:3] # rotation matrix in SO3

        A = np.zeros((2,2)) # state matrix
        B = np.array([[0,1,0],[0,0,1]]) @ R # input matrix
        E = np.linalg.inv(cls.X) @ cls.X # group error element

        # define the global error:
        e = cls.phi(np.linalg.inv(cls.X), gb)
        # define the local coordinates of the global error:
        epsilon = cls.theta(e) 
        eps_wedge = se3_wedge(epsilon)

        # define the measurements y
        g = gb / np.absolute(gb)
        ghat = cls.ghat # updated in the filtering loop 

        # equivariant output matrix
        arr = np.array([[0, 0],[1,0],[0,1]])
        ys = se3_wedge(g) + se3_wedge(ghat)
        Cts = 0.5 * (ys) @ R.T @ arr

        Dt = np.eye(3) # jacobian of the log map at identity (theta derivatiuve)
        # compute state and output gain matrices
        M = B @ cls.Q**3 @ B.T + B @ cls.Q @ B.T
        N = Dt @ cls.Rcov @ Dt.T

        # UPDATE EVERYTHING =================================================
        # THE OUTPUTS ARE TIME DERIVATIVES AND WILL BE NUMERICALLY INTEGRATED
        # ricatti termm
        cls.sigma = A @ cls.sigma + cls.sigma @ A.T + M - (cls.sigma @ Cts.T @ np.linalg.inv(N) @ Cts @ cls.sigma)
        # kalman gain
        residual = gb - cls.phi(cls.X, cls.xi)
        Delta = -se3_wedge(cls.xi0) @ (np.eye(3) - cls.xi0 @ cls.xi0.T) @ cls.sigma @ Cts.T @ np.linalg.inv(N) @ residual
        # update the observers state Xhat.
        Xdot = se3_wedge(ghat) @ cls.X + Delta @ R

        return Xdot
